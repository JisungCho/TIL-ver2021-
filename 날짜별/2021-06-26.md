### 플러스 원(Plus One)

1. 문제분석
   1. 뒷 자리부터 체크
   2. 값이 10이되면 carry를 1로
   3. 맨마지막에 carry가 일어나면 확인

```java

    //새로운 방법
	public static int[] solve(int[] digits) {
		int n = digits.length;

		for (int i = n - 1; i >= 0; i--) {
			digits[i]++;

			if (digits[i] < 10) {
				return digits;
			}

			digits[i] = 0;
		}

		int[] res = new int[n + 1];
		res[0] = 1;

		return res;
	}
	
    //기존방법
	public static int[] plusOne2(int[] digits) {
		int carry = 1;
		int index = digits.length - 1;
		while (index >= 0 && carry > 0) {
			digits[index] = (digits[index] + carry) % 10;

			if (digits[index] == 0) {
				carry = 1;
			} else {
				carry = 0;
			}
			index--;
		}

		if (carry > 0) { //맨 마지막 자리에서 carry발생
			digits = new int[digits.length + 1];
			digits[0] = 1;
		}

		return digits;
	}
}
```
