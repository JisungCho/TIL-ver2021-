```java
package sec01;

import java.util.Scanner;

class Main {
	public static int solution(int n , int[] arr){
		int cnt = 0;
		for(int i=2;i<n+1;i++) {
			if(arr[i] == 0) {
				System.out.println(i);
				cnt++;
				for(int j=i;j<n+1;j=j+i) {
						arr[j] = 1;
				}
			}
		}
		return cnt;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt();
		int[] arr = new int[n+1];
		
		System.out.println(solution(n,arr));
	}
}
```

```java
package sec01;

import java.util.ArrayList;
import java.util.Scanner;

class Main {
	public static boolean isPrime(int num){
		if(num==1) return false;
		for(int i=2; i<num; i++){
			if(num%i==0) return false;
		}
		return true;
	}
	public static ArrayList<Integer> solution(int n , int[] arr){
		ArrayList<Integer> answer = new ArrayList<>();
		for(int i=0; i<n; i++){
			int tmp=arr[i];
			int res=0;
			while(tmp>0){
				int t=tmp%10;
				res=res*10+t;
				tmp=tmp/10;
			}
			if(isPrime(res)) answer.add(res);
		}
		return answer;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt();
		int[] arr = new int[n];
		for(int i =0;i<n;i++) {
			arr[i] = kb.nextInt();
		}
		System.out.println(solution(n,arr));
	}
}
```

### 점수 계산

1. 문제분석
   - 1이나오면 total에 point를 누적
   - 0을 만나면 point를 1로만들어주고 더해주지않음

```java
package sec01;

import java.util.ArrayList;
import java.util.Scanner;

class Main {
	public static int solution(int n , int[] arr){
		int total = 0;
		int point = 1;
		for(int i=0;i<n;i++) {
			if(arr[i] == 1) {
				total += point;
				point++;
			}else {
				point = 1;
			}
		}
		return total;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt();
		int[] arr = new int[n];
		for(int i =0;i<n;i++) {
			arr[i] = kb.nextInt();
		}
		System.out.println(solution(n,arr));
	}
}
```

### 등수구하기

1. 문제분석
   1. 이중for사용
   2. cnt는 등수

```java
import java.util.*;
class Main {	
	public int[] solution(int n, int[] arr){
		int[] answer = new int[n];
		for(int i=0; i<n; i++){
			int cnt=1;
			for(int j=0; j<arr.length; j++){
				if(arr[j]>arr[i]) cnt++;
			}
			answer[i]=cnt;
		}
		return answer;
	}
	public static void main(String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		int n=kb.nextInt();
		int[] arr=new int[n];
		for(int i=0; i<n; i++){
			arr[i]=kb.nextInt();
		}
		for(int x :T.solution(n, arr)) System.out.print(x+" ");
	}
}
```
