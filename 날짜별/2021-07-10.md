## Array

### 큰 수 출력하기

```java
package sec01;

import java.util.ArrayList;
import java.util.Scanner;

class Main {
	public static ArrayList<Integer> solution(int[] arr,int n){
		ArrayList<Integer> answer = new ArrayList<>();
		int prev = arr[0];
		answer.add(arr[0]);
		
		for(int i=1;i<n;i++) {
			if(prev < arr[i]) {
				answer.add(arr[i]);
			}
			prev = arr[i];
		}
		
		return answer;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt();
		int[] arr = new int[n];
		for(int i=0;i<n;i++) {
			arr[i] = kb.nextInt();
		}
		for(int i : solution(arr,n)) {
			System.out.print(i+" ");
		}
	}
}
```

### 보이는 학생

```java
package sec01;

import java.util.ArrayList;
import java.util.Scanner;

class Main {
	public static int solution(int[] arr){
		int answer = 0;
		int max = Integer.MIN_VALUE;
		for(int i : arr) {
			if(max < i) {
				max = i;
				answer++;
			}
		}
		
		return answer;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt();
		int[] arr = new int[n];
		for(int i=0;i<n;i++) {
			arr[i] = kb.nextInt();
		}
		System.out.println(solution(arr));
	}
}
```
