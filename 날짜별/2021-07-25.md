### 연속 부분수열

```java
package sec01;

import java.util.Scanner;

class Main {
	public static int solution(int n,int m ,int[] arr){
		int answer = 0;
		int lt=0;
		int rt =0;
		int sum = 0;
		
		while(rt < n) {
			sum += arr[rt]; //rt의 값을 계속 더해줌
			if(sum == m) answer++; //같으면 cnt++;
			while(sum>=m) { // sum이 m보다 크거나 같을때까지 
				sum-=arr[lt]; // lt값을 빼주고
				lt++; // lt인덱스 증가
				if(sum == m)answer++;
			}
			rt++;
		}
		
		return answer;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt(); // 원소의 갯수
		int m = kb.nextInt(); // 부분수열의 합
		int[] arr =new int[n];
		for(int i=0;i<n;i++) { // 원소의 값
			arr[i] = kb.nextInt();
		}

		System.out.println(solution(n, m, arr));
	}
}
```

### 연속된 자연수의 합

		int lt=1;
		int rt =1;
		int sum = 0;
		
		while(rt < n) {
			sum += rt; //rt 더해줌
			if(sum == n) answer++; //같으면 cnt++;
			while(sum>=n) { // sum이 m보다 크거나 같을때까지 
				sum-=lt; // lt 빼주고
				lt++; // lt 증가
				if(sum == n)answer++;
			}
			rt++;
		}
		
		return answer;
	}

	public static void main(String[] args){
		Scanner kb = new Scanner(System.in);
		
		int n = kb.nextInt(); // 자연수

		System.out.println(solution(n));
	}
}

### 최대 길이 연속부분수열

```java
package sec01;

import java.util.Scanner;

class Main {
	public static int solution(int n, int k, int[] arr) {
		int answer=0, cnt=0, lt=0;
		for(int rt=0; rt<n; rt++){
			if(arr[rt]==0) cnt++;
			while(cnt>k){
				if(arr[lt]==0) cnt--;
				lt++;
			}
			answer=Math.max(answer, rt-lt+1);
		}
		return answer;
	}

	public static void main(String[] args) {
		Scanner kb = new Scanner(System.in);

		int n = kb.nextInt(); // 수열의 길이
		int k = kb.nextInt(); // 바꿀 수 있는 횟수
		int[] arr = new int[n];
		for (int i = 0; i < n; i++) {
			arr[i] = kb.nextInt();
		}

		System.out.println(solution(n, k, arr));
	}
}
```
